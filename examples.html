Hello, World

The minimal Forge program. Demonstrates the entry point, a string literal, a stack-allocated value, and the built-in print function.

main.forge
// Every Forge program starts at `main`.
// No return type = implicitly returns nothing (unit).
fn main() {
    print("Hello, World!");
}
Output
$ forge run main.forge
Hello, World!
📌
No region annotation needed. String literals are @static — they live for the program's lifetime. The compiler infers this; you never write it.
📌
No heap allocation. Nothing dynamic happens here. The string is a pointer into read-only program memory.
extended — variables and formatted output
fn main() {
    // Type annotation omitted — defaults to str@static
    let lang = "Forge";

    // i32 with no region annotation = i32@stack
    let year: i32 = 2025;

    print("Hello from {} ({})", lang, year);

    // Mutable stack variable
    let count: i32 = 0;
    count += 1;
    print("count is now {}", count);
}
Output
Hello from Forge (2025)
count is now 1

Factorial

Two implementations — recursive and iterative — showing function definitions, branching, and basic arithmetic. Also demonstrates passing values vs. pointers.

factorial_recursive.forge
// Recursive factorial.
// Parameters with no region = @stack by default.
fn factorial(n: u64) -> u64 {
    if n <= 1 {
        return 1;
    }
    return n * factorial(n - 1);
}

fn main() {
    let n: u64 = 10;
    let result = factorial(n);   // n is copied — u64 is a value type
    print("{}! = {}", n, result);
}
Output
10! = 3628800
factorial_iterative.forge
// Iterative factorial with a mutable accumulator.
fn factorial(n: u64) -> u64 {
    let result: u64 = 1;   // u64@stack
    let i: u64      = 2;

    loop {
        if i > n { break; }
        result *= i;
        i += 1;
    }

    return result;
}

// Accumulate a table of factorials, stored on the heap.
fn factorial_table(max: u64) -> [u64]@heap {
    let table: [u64]@heap = alloc(u64, max + 1);
    let i: u64 = 0;
    loop {
        if i > max { break; }
        table[i] = factorial(i);
        i += 1;
    }
    return table;   // ownership moves to caller — no copy
}

fn main() {
    let tbl = factorial_table(10);   // owns [u64]@heap
    let i: u64 = 0;
    loop {
        if i > 10 { break; }
        print("{}! = {}", i, tbl[i]);
        i += 1;
    }
    destroy tbl;   // explicit dealloc — heap memory freed here
}
Output
0! = 1
1! = 1
2! = 2
3! = 6
4! = 24
5! = 120
6! = 720
7! = 5040
8! = 40320
9! = 362880
10! = 3628800
📌
Return transfers ownership. factorial_table returns a [u64]@heap. The heap allocation is not copied — its ownership moves to the caller. The function's local binding table is invalidated on return.
📌
Explicit destroy. Heap values must be explicitly destroy-ed or moved. The compiler will warn on any heap value that goes out of scope without being destroyed or returned.

GCD Calculator

Euclidean algorithm in two forms. Also shows a mutable-pointer version that updates a result in place, demonstrating *T borrow semantics.

gcd.forge
// Classic Euclidean GCD — pure value semantics.
// Both parameters are u64@stack (default).
fn gcd(a: u64, b: u64) -> u64 {
    loop {
        if b == 0 { return a; }
        let tmp = b;
        b = a % b;
        a = tmp;
    }
}

// LCM derived from GCD — shows composing pure functions.
fn lcm(a: u64, b: u64) -> u64 {
    return (a / gcd(a, b)) * b;   // divide first to avoid overflow
}

fn main() {
    let pairs: [u64] = [48, 18,  100, 75,  0, 7,  13, 7];
    let i: u64 = 0;

    loop {
        if i >= 8 { break; }
        let a = pairs[i];
        let b = pairs[i + 1];
        print("gcd({}, {}) = {}  lcm = {}", a, b, gcd(a, b), lcm(a, b));
        i += 2;
    }
}
Output
gcd(48, 18) = 6 lcm = 144
gcd(100, 75) = 25 lcm = 300
gcd(0, 7) = 7 lcm = 0
gcd(13, 7) = 1 lcm = 91
gcd_inplace.forge — pointer receiver demo
// In-place version: result is written through a mutable pointer.
// Shows *T@stack borrow — caller keeps ownership of `out`.
fn gcd_into(a: u64, b: u64, out: *u64) {
    loop {
        if b == 0 { *out = a; return; }
        let tmp = b;
        b = a % b;
        a = tmp;
    }
}

fn main() {
    let result: u64 = 0;   // u64@stack — lives here

    gcd_into(56, 98, &result); // mutable borrow of `result`
                               // borrow ends after this call
    print("gcd = {}", result);  // ✓ result is valid — borrow is over
}
Output
gcd = 14
📌
Borrow ends at last use. The mutable borrow &result passed to gcd_into is dead the moment the call returns. The print on the next line reads result directly — no conflict.
📌
*out = a dereferences to write. Inside gcd_into, out is a *u64@stack. Writing through it with *out = a is a normal mutable-pointer write — no unsafe block needed.

Structs & Interfaces

A geometry system with two shapes, a shared Shape interface, Go-style method receivers, and both static and dynamic dispatch.

shapes.forge — type definitions
// ── Structs ──────────────────────────────────────────

struct Circle {
    radius: f64,
}

struct Rect {
    width:  f64,
    height: f64,
}

// ── Interface ─────────────────────────────────────────
// Any type with `area` and `perimeter` methods satisfies Shape.
// No `implements` keyword — satisfaction is implicit.

interface Shape {
    fn area(self) -> f64;
    fn perimeter(self) -> f64;
    fn describe(self) -> str;
}

// ── Methods on Circle ─────────────────────────────────
// Value receiver: `self` is a copy of the Circle.

fn (Circle) area(self) -> f64 {
    return 3.14159265 * self.radius * self.radius;
}

fn (Circle) perimeter(self) -> f64 {
    return 2.0 * 3.14159265 * self.radius;
}

fn (Circle) describe(self) -> str {
    return fmt("Circle(r={})", self.radius);
}

// Pointer receiver: mutates the Circle in place.
fn (*Circle) scale(self, factor: f64) {
    self.radius *= factor;
}

// ── Methods on Rect ───────────────────────────────────

fn (Rect) area(self) -> f64 {
    return self.width * self.height;
}

fn (Rect) perimeter(self) -> f64 {
    return 2.0 * (self.width + self.height);
}

fn (Rect) describe(self) -> str {
    return fmt("Rect({}x{})", self.width, self.height);
}

fn (*Rect) scale(self, factor: f64) {
    self.width  *= factor;
    self.height *= factor;
}
shapes.forge — static dispatch via generic constraint
// T must satisfy Shape — resolved at compile time (monomorphized).
// Separate compiled versions for Circle and Rect — zero overhead.
fn print_info<T: Shape>(s: T) {
    print("{}",          s.describe());
    print("  area      = {:.4}", s.area());
    print("  perimeter = {:.4}", s.perimeter());
}

// Constraint can also be a type constraint group.
// `Scalable` is any type that has both Shape methods and scale().
interface Scalable {
    fn area(self)      -> f64;
    fn perimeter(self) -> f64;
    fn describe(self)  -> str;
    fn scale(*self, factor: f64);  // pointer receiver required
}

fn scale_and_print<T: Scalable>(s: *T, factor: f64) {
    s.scale(factor);                       // mutates in place
    print("{} after scale({})", s.describe(), factor);
    print("  new area = {:.4}", s.area());
}
shapes.forge — dynamic dispatch via interface value
// `Shape` used as a concrete type = dynamic dispatch (vtable).
// `shapes` is a slice of interface values — each may be a different type.
fn total_area(shapes: [Shape]) -> f64 {
    let sum: f64 = 0.0;
    let i: u64   = 0;
    loop {
        if i >= len(shapes) { break; }
        sum += shapes[i].area();   // vtable call — type resolved at runtime
        i += 1;
    }
    return sum;
}
shapes.forge — main
fn main() {
    // Stack-allocated structs — no @heap needed for simple local use.
    let c: Circle = Circle{ .radius = 5.0 };
    let r: Rect   = Rect  { .width = 4.0, .height = 6.0 };

    // Static dispatch — compiler generates Circle-specific code.
    print_info(c);
    print_info(r);

    print("---");

    // Pointer receiver — scale() mutates in place.
    // Compiler auto-takes &c and &r (auto-reference rule).
    scale_and_print(&c, 2.0);
    scale_and_print(&r, 0.5);

    print("---");

    // Dynamic dispatch — mixed slice of Shape interface values.
    let shapes: [Shape] = [c, r];   // c and r are boxed into interface values
    print("total area = {:.4}", total_area(shapes));

    let i: u64 = 0;
    loop {
        if i >= len(shapes) { break; }
        print("  {}", shapes[i].describe());
        i += 1;
    }
}
Output

area = 78.5398
perimeter = 31.4159
Rect(4x6)
area = 24.0000
perimeter = 20.0000
---
Circle(r=10) after scale(2)
new area = 314.1593
Rect(2x3) after scale(0.5)
new area = 6.0000
---

Circle(r=10)
Rect(2x3)
📌
No implements declaration. Circle and Rect satisfy Shape automatically because they have the right methods. The compiler checks this at the call site.
📌
Static vs. dynamic dispatch. print_info<T: Shape> is monomorphized — one compiled copy per concrete type, zero vtable overhead. total_area([Shape]) uses a vtable — necessary because the slice holds mixed types.
📌
Auto-reference on pointer receivers. scale_and_print(&c, 2.0) passes a *Circle@stack. Inside the function, calling s.scale(factor) routes to fn (*Circle) scale — no explicit dereference needed.
📌
Stack allocation is the default. c and r live entirely on the stack. No heap allocation occurs until they are boxed into the [Shape] interface slice.
